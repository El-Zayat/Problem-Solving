int dfs(TreeNode* &node, int mx, int mn) {
    if(node == nullptr) return 0;
    mx = max(mx, node->val);
    mn = min(mn, node->val);
    return max({mx - mn, dfs(node->left, mx, mn), dfs(node->right, mx, mn)});
}

int maxAncestorDiff(TreeNode* &root) {
    return dfs(root, root->val, root->val);
}
