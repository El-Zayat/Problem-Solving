let mod = n => {
   let m = 1e9 + 7;
   return ((n%m)+m)%m;
}

let kInversePairs = function(n, k) {

   // dp[i][o] is how many permutations of i numbers that have o inversions
   
   let dp = [[1], [1]];

   for(let i = 1; i <= n; ++i) {
       let it = i&1;

       for(let o = 1; o <= k; ++o) {
           dp[it^1][o] = mod((dp[it^1][o] || 0) + (dp[it^1][o-1] || 0))
           dp[it][o] = mod(dp[it^1][o] - (o-i>=0?dp[it^1][o-i] : 0));
       }
   }

   return dp[n&1][k];
};
