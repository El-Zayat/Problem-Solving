let evalRPN = function (tokens) {
    let ops = ['/', '*', '+', '-'];

    let process = (op, a, b) => {
        switch (op) {
            case '+':
                return a + b;
            case '-':
                return a - b;
            case '*':
                return a * b;
            case '/':
                return a / b < 0 ? Math.ceil(a / b) : Math.floor(a / b);
        }
    }

    let st = [];

    for (let t of tokens) {
        if (ops.includes(t)) {
            let b = st.pop(), a = st.pop();
            st.push(process(t, a, b));
        } else st.push(parseInt(t));
    }

    return st[0];
};
