```js
let findPaths = function(m, n, rem, i, j, dp = []) {
    if(i<0 || i==m || j<0 || j==n) return 1;
    if(!rem) return 0;

    dp[i] = dp[i] || [];
    dp[i][j] = dp[i][j] || [];

    if(rem in dp[i][j])
        return dp[i][j][rem];

    dp[i][j][rem]  = findPaths(m, n, rem-1, i+1, j, dp);
    dp[i][j][rem] += findPaths(m, n, rem-1, i-1, j, dp);
    dp[i][j][rem] += findPaths(m, n, rem-1, i, j+1, dp);
    dp[i][j][rem] += findPaths(m, n, rem-1, i, j-1, dp);

    dp[i][j][rem] %= 1e9 + 7;

    return dp[i][j][rem];
};
```
