let findDuplicate = function(nums) {
    let n = nums.length;
    let l = 1, r = n, mid, res;
    while(l <= r) {
        mid = l + Math.floor((r-l)/2);

        let cnt = 0;
        for(let i of nums)
            cnt += i <= mid;
        
        if(cnt > mid) {
            res = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    return res;
};
