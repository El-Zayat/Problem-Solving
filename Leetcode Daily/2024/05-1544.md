let makeGood = function(s) {
    let st = [];
    let isUpper = c => c == c.toUpperCase();

    for(let c of s) {
        if(st.length && (isUpper(c) != isUpper(st[st.length - 1])) && c.toLowerCase() === st[st.length - 1].toLowerCase())
            st.pop();
        else
            st.push(c);
    }

    return st.join('');
};
