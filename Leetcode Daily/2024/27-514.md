int findRotateSteps(string ring, string key) {
    vector<vector<int>> ind(26);
    int n = key.size();
    int m = ring.size();

    for(int i = 0; i < m; ++i)
        ind[ring[i] - 'a'].push_back(i);

    int dp[2][m];
    for(int i = 0; i < m; ++i)
        dp[n&1][i] = 0;

    const int oo = 1e9;
    for(int i = n-1; i >= 0; --i) {
        int it = i&1;
        for(int j = 0; j < m; ++j) {
            dp[it][j] = oo;
            for(int k : ind[key[i] - 'a']) {
                dp[it][j] = min(dp[it][j], dp[it^1][k] + min({abs(k-j), j + m - k, k + m - j}) + 1);
            }
        }
    }

    return dp[0][0];
}
