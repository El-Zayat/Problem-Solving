let productExceptSelf = function(nums) {
    let n = nums.length;
    let pre = new Array(n+1).fill(1), suff = new Array(n + 2).fill(1);
    
    for(let i = 1, j = n; i <= n; ++i, --j) {
        pre[i] = pre[i-1] * nums[i - 1];
        suff[j] = suff[j + 1] * nums[j - 1];
    }
    
    return nums.map((a, i) => pre[i]*suff[i+2]);
};
