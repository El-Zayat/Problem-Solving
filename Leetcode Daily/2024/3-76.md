let minWindow = function(s, t) {
    let f = {};
    let cnt = 0;

    for(let c of t) {
        cnt += !f[c];
        f[c] = f[c]+1 || 1;
    }

    let l = 0, r = -1;
    let L = -1, R = 2e5;

    while(r<s.length) {
        if(cnt) {
            if(++r < s.length && f[s[r]] != undefined && --f[s[r]] === 0)
                cnt--;
        } else {
            if(f[s[l]] != undefined && f[s[l]]++ == 0)
                cnt++;
            ++l;
        }

        if(!cnt && r - l < R - L)
            L = l, R = r;
    }

    if(L != -1)
        return s.substring(L, R+1);
    return "";
};
