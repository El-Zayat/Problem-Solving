vector<vector<int>> findMatrix(vector<int>& nums) {
    int n = nums.size();
    vector<vector<int>> res;
    vector<int> f(n + 1);
    vector<bool> vis(n);
    res.push_back({});

    int cnt = 0;
    while(cnt < n) {
        for(int i = 0; i < n; ++i) {
            if(!f[nums[i]] && !vis[i]) {
                f[nums[i]]++;
                vis[i] = 1;
                res.back().push_back(nums[i]);
                ++cnt;
            }
        }
        res.push_back({});
        f.assign(n + 1, 0);
    }
    res.pop_back();
    return res;
}
