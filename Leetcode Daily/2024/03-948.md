let bagOfTokensScore = function(tokens, power) {
    tokens.sort((a, b) => b - a);

    let score = 0;
    let res = 0;
    let l = 0, r = tokens.length - 1;
    while(l <= r) {
        if(tokens[r] <= power) {
            score++;
            power -= tokens[r--];
        } else if(score) {
            power += tokens[l++];
            score--;
        } else break;

        res = Math.max(res, score);
    }

    return res;
};
