let groupAnagrams = function(strs) {
   let map = {};
   for(let str of strs) {
       let sorted = str;
       sorted = sorted.split('').sort().join('');
       if(sorted in map)
            map[sorted].push(str);
        else
            map[sorted] = [str];
   }

   return Object.values(map);
};
