```js

var MyQueue = function() {
    this.a = [];
};

MyQueue.prototype.push = function(x) {
    this.a.push(x);
};

MyQueue.prototype.pop = function() {
    let b = [];

    while(this.a.length)
        b.push(this.a.pop());

    let p = b.pop();

    while(b.length)
        this.a.push(b.pop());
    
    return p;
};

MyQueue.prototype.peek = function() {
    let b = [];
    while(this.a.length > 1)
        b.push(this.a.pop());    

    let p = this.a[0];
    
    while(b.length)
        this.a.push(b.pop());
    return p;
};

MyQueue.prototype.empty = function() {
    return !this.a.length;
};
```
