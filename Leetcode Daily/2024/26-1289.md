class Solution {
private:
    int n, id = 0;
    int dp[201][201], vis[201][201];

    int go(int r, int p, auto &v) {
        if(r == n) return 0;

        int &ret = dp[r][p];
        if(vis[r][p] == id)
            return ret;
        vis[r][p] = id;

        ret = 1e9;
        for(int i = 0; i < n; ++i)
            if(i != p)
                ret = min(ret, v[r][i] + go(r+1, i, v));
        
        return ret;
    }
public:
    int minFallingPathSum(vector<vector<int>>& grid) {
        n = grid.size();
        ++id;
        return go(0, n, grid);
    }
};
