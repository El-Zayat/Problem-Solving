let minFallingPathSum = function(matrix) {
    let n = matrix.length;
    let dp = new Array(n);
    
    for(let i = n-1; i >= 0; --i) {
        dp[i] = [];
        for(let j = 0; j < n; ++j) {
            dp[i][j] = matrix[i][j];
            if(i == n-1) continue;
            
            let add = dp[i+1][j];

            if(j)
                add = Math.min(add, dp[i+1][j-1]);
            if(j+1 < n)
                add = Math.min(add, dp[i+1][j+1]);

            dp[i][j] += add;
        }
    }
    
    let res = 2e9;
    for(let i = 0; i < n; ++i)
        res = Math.min(dp[0][i], res);
    
    return res;
};
