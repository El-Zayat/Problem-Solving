let closeStrings = function(a, b) {
    if(a.length !== b.length) return false;

    let freqB = {}, freqA = {};

    for(let c of a)
        freqA[c] = freqA[c] + 1 || 1;

    for(let c of b)
        freqB[c] = freqB[c] + 1 || 1;

    let cntA = [], cntB = [];
    for(let c of "abcdefghijklmnopqrstuvwxyz") {
        if(!freqA[c] && !freqB[c]) continue;
        if(!freqA[c] || !freqB[c]) return false;

        cntA.push(freqB[c]);
        cntB.push(freqA[c]);
    }

    cntA.sort((a, b) => a - b);
    cntB.sort((a, b) => a - b);
    return cntA.every((a, i) => a == cntB[i]);
};
