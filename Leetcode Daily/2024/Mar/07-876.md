let middleNode = function(head) {
    let getSz = (u) => {
        if(!u) return 0;
        return 1 + getSz(u.next);
    };

    let mid = Math.floor(getSz(head)/2) + 1;
    
    let move = (u, o) => {
        if(o === mid)
            return u;
        return move(u.next, o+1);
    };

    return move(head, 1);
};
