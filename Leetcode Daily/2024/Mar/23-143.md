let reorderList = function(head) {
    let a = [];
    
    let go = (u) => {
        if(!u) return;
        a.push(u);
        go(u.next);
    }
    go(head);


    if(a.length&1)
        a[Math.floor(a.length / 2)].next = null;

    for(let i = 0, j = a.length - 1; i < j; i++, j--) {
        a[i].next = a[j];
        a[j].next = i+1 == j? null : a[i + 1];
    }

};
