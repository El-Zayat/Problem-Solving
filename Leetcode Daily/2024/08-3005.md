let maxFrequencyElements = function(v) {
    let mx = 0;
    let rep = 0;

    let f = {};
    for(let i of v) {
        f[i] = f[i]+1 || 1;
        if(mx<f[i]) {
            mx = f[i];
            rep = 1;
        }
        else if(mx == f[i])
            rep++;
    }
    

    return mx * rep;
};
