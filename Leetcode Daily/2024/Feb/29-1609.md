let isEvenOddTree = function(root) {
    let lvl = [];
    
    let dfs = (u, l) => {
        if(!u) return;
        lvl[l] = lvl[l] || [];
        lvl[l].push(u.val);
        dfs(u.left, l+1);
        dfs(u.right, l+1);
    }

    dfs(root, 0);

    for(let i = 0; i < lvl.length; ++i) {
        if((lvl[i][0] & 1) === (i&1)) return false;
        for(let j = 1; j < lvl[i].length; ++j){
            let v = lvl[i][j];
            if(i&1) {
                if(v&1) return false;
                if(v >= lvl[i][j-1]) return false;
            }
            else {
                if(!(v&1)) return false;
                if(v <= lvl[i][j-1]) return false;
            }
        }
    }

    return true;
};
