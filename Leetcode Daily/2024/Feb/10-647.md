let countSubstrings = function(s) {
  let n = s.length;

  let cnt = 0;
  for(let i = 0; i < n; ++i) {
    for(let j = i, o = i; j >= 0 && o < n; --j, ++o) {
      if(s[j] != s[o])
        break;
      cnt++
    }

    for(let j = i, o = i+1; j >= 0 && o < n; --j, ++o) {
      if(s[j] != s[o])
        break;
      cnt++
    }
  }

  return cnt;
};
