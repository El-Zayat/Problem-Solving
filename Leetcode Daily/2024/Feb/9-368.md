let largestDivisibleSubset = function(a) {
  a.sort((b, c) => c - b);
  let n = a.length;
  let dp = new Array(n).fill([1, -1]);

  let res = dp[0];
  let ret = [a[0]];

  for(let i = 1; i < n; ++i) {
    for(let j = i - 1; j >= 0; --j)
      if(!(a[j] % a[i]) && dp[j][0]+1 > dp[i][0])
        dp[i] = [dp[j][0]+1, j];
    
    if(dp[i][0] > res[0]) {
      res = dp[i];
      ret = [a[i]];
    }
  }

  let nxt = res[1];
  while(~nxt) {
    ret.push(a[nxt]);
    nxt = dp[nxt][1];
  }

  return ret;
}
